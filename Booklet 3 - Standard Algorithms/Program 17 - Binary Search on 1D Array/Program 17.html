def binarySearch(nA, target):
    low = 0
    high =  len(nA)-1
    mid = 0
    found = False
    position = -1

    while not(found) and low <= high:
        mid = int((low + high)/2)
        if target == nA[mid]:
            position = mid
            found = True
        elif target >= nA[mid]:
            low = mid + 1
        else:
            high = mid - 1     
    
    return position
